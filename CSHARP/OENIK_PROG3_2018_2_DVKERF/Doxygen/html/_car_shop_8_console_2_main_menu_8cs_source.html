54  public virtual void SelectMenuItem()
55  {
56  Console.WriteLine("Select a Menu by pressing the relevant command");
57 
58  // gets input from the user
59  string command = Console.ReadLine().ToUpper();
60 
61  // selects a SubMenu
62  var selected = this.SubMenus.Find(itm => itm.Command == command);
63 
64  // if the command is E the private field "exit" will be set to true, this will break the StartMenu() loop
65  if (command == "E")
66  {
67  this.exit = true;
68  Console.WriteLine("Exiting application.....");
69  Console.ReadLine();
70  }
71 
72  // checks if a valid Sub Menu was selected
73  if (selected != null)
74  {
75  // sets the SubMenu's Parrent Item to 'this'. This will help the SubMenu to go Back to the Main Menu.
76  selected.ParrentMenu = this;
77  selected.StartSubMenu();
78  }
79  else if (selected == null && command != "E")
80  {
81  // If an invalid command was received the method will call itself again.
82  Console.WriteLine("Invalid Command, Press Any key to try again!");
83  Console.ReadLine();
84  this.DisplayMenu();
85  this.SelectMenuItem();
86  }
87  }
