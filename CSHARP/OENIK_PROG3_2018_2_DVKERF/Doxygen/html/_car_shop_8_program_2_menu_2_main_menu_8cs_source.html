62  public void SelectMenuItem()
63  {
64  Console.WriteLine("Press a Button");
65  string command = Console.ReadLine();
66  MenuItem menuItem = this.MenuItems.Find(x => x.Command == command.ToUpper());
67 
68  if (menuItem != null)
69  {
70  if (menuItem.MenuAction.GetType() == typeof(ExitAction))
71  {
72  this.ExitMenu = true;
73  }
74 
75  menuItem.MenuAction.ExecuteMenuAction();
76  }
77  else
78  {
79  Console.Clear();
80  Console.WriteLine("Invalid Command, please enter another key");
81  }
82  }
